-module(blog).
-export([main/1, run/1]).

run(BlogDir) ->
    main(BlogDir),
    io:format("ok~n"),
    halt(0).

main(BlogDir) ->
    {Src, Out, Index, Vars} = load_conf(BlogDir),
    Pages = [{D, T, sluggify(T), Txt} || {D, T, Txt} <- compile_files(Index, Vars)],
    recursive_copy(Src, Out),
    save_pages(Out, Pages),
    create_index(Src, Out, Pages, Vars),
    ok.
